// Main commands

// Build the images

docker build -t service-discovery . 
docker build -t traffic-analytics-service-1 .
docker build -t traffic-analytics-service-2 . 
docker build -t traffic-analytics-service-3 .  
docker build -t traffic-regulation-service-1 .
docker build -t traffic-regulation-service-2 . 
docker build -t traffic-regulation-service-3 .  
docker build -t traffic-analytics-load-balancer . 
docker build -t traffic-regulation-load-balancer . 
docker build -t gateway . 

// Tag and push images to Docker Hub

docker tag image:latest username/image:latest
docker push username/image:latest

// Build the project using Docker Compose

docker-compose up --build 


// Additional commands

// Remove all containers

docker stop $(docker ps -a -q)
docker rm $(docker ps -a -q)

// Remove all images

docker rmi $(docker images -a -q)
